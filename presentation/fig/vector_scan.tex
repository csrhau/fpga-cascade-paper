\documentclass[tikz]{standalone}
\usetikzlibrary{patterns}

\begin{document}

  \definecolor{printred}{RGB}{215,25,28}
  \definecolor{printgreen}{RGB}{145,180,130}
  \tikzset{>=latex} % Change default arrowhead to filled triangle
  \begin{tikzpicture}[] % scale=0.25]
    \pgfmathsetmacro{\domainwidth}{5}
    \pgfmathsetmacro{\domainheight}{4}
    \pgfmathsetmacro{\memorywidth}{\domainwidth+2}
    \pgfmathsetmacro{\memoryheight}{\domainheight+2}

    \begin{overprint}
    
    \draw[gray, step=1] (0,0) grid (\memorywidth,\memoryheight);

    \draw[printred, densely dashed, very thick] (1, 0) rectangle (\domainwidth+1, 1);
    \draw[printred, densely dashed, very thick] (1, \domainheight+1) rectangle (\domainwidth+1, \domainheight+2);
    \draw[printred, densely dashed, very thick] (0, 1) rectangle (1, \domainheight+1);
    \draw[printred, densely dashed, very thick] (\domainwidth+1, 1) rectangle (\domainwidth+2, \domainheight+1);
    \draw[darkgray, very thick](1, 1) rectangle (\domainwidth+1,\domainheight+1);


    \foreach \ts in {1,...,\domainheight} {
      % From hatching
      \onslide<+> {
        \draw[pattern color=lightgray, pattern=crosshatch, very thin](1, \domainheight+1 - \ts) rectangle (\domainwidth+1, \domainheight+2-\ts);
        \draw[dashed, thick, ->] (1.5, \domainheight+1.5-\ts) -- (2.5, \domainheight+2.5-\ts); 
        \draw[dashed, thick, ->] (0.5+\domainwidth, \domainheight+1.5-\ts) -- (1.5+\domainwidth, \domainheight+2.5-\ts); 
      }

      % To hatching
      \onslide<\ts-> {
        \draw[pattern color=lightgray, pattern=north west lines, very thin](2, \domainheight+2-\ts) rectangle (\domainwidth+2, \domainheight+3-\ts);
      }
    }
    \onslide<+>{}

    \end{overprint}

  
   
  \end{tikzpicture}
\end{document}
