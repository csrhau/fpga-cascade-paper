\documentclass[tikz]{standalone}
\usetikzlibrary{patterns}
\begin{document}
  \def\stencil[#1, #2, #3]{%
    \draw[very thick, #3] (#1, #2) rectangle (#1+3, #2+3);
    \draw[very thick, densely dashed, #3] (#1-1, #2) rectangle (#1, #2+3);
    \draw[very thick, densely dashed, #3] (#1+3, #2) rectangle (#1+4, #2+3);
    \draw[very thick, densely dashed, #3] (#1, #2-1) rectangle (#1+3, #2);
    \draw[very thick, densely dashed, #3] (#1, #2+3) rectangle (#1+3, #2+4);
  }

  \begin{tikzpicture}[scale=0.5]
    \draw[gray, step=1] (0,0) grid (8,8);
    \stencil[2, 2, red]
    \stencil[3, 3, blue]
  \end{tikzpicture}
\end{document}
